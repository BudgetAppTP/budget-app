% GLOSSARY ENTRIES

\newacronym{api}{API}{Application Programming Interface -- aplikačné programové rozhranie}
\newacronym{ci}{CI}{Continuous Integration -- kontinuálna integrácia}
\newacronym{cors}{CORS}{Cross-Origin Resource Sharing -- zdieľanie zdrojov medzi doménami}
\newacronym{csv}{CSV}{Comma-Separated Values -- súborový formát na ukladanie tabuľkových dát}
\newacronym{db}{DB}{Database -- databáza}
\newacronym{dom}{DOM}{Document Object Model -- objektový model dokumentu}
\newacronym{dto}{DTO}{Data Transfer Object -- objekt na prenos dát medzi vrstvami systému}
\newacronym{eslint}{ESLint}{-- nástroj na statickú analýzu kódu}
\newacronym{http}{HTTP}{Hypertext Transfer Protocol -- hypertextový prenosový protokol}
\newacronym{https}{HTTPS}{HTTP Secure -- zabezpečený hypertextový prenosový protokol}
\newacronym{id}{ID}{Identifier -- identifikátor}
\newacronym{json}{JSON}{JavaScript Object Notation -- formát na výmenu dát}
\newacronym{jwt}{JWT}{JSON Web Token -- štandard pre bezpečný prenos informácií}
\newacronym{orm}{ORM}{Object-Relational Mapping -- objektovo-relačné mapovanie}
\newacronym{pdf}{PDF}{Portable Document Format -- prenosný formát dokumentov}
\newacronym{pos}{POS}{Point of Sale -- miesto predaja (pokladničný systém)}
\newacronym{qr}{QR}{Quick Response Code -- 2D kód pre rýchle načítanie informácií}
\newacronym{rest}{REST}{Representational State Transfer -- architektonický štýl pre návrh webových API}
\newacronym{spa}{SPA}{Single-Page Application -- jednostránková aplikácia}
\newacronym{tls}{TLS}{Transport Layer Security -- zabezpečenie transportnej vrstvy}
\newacronym{ui}{UI}{User Interface -- používateľské rozhranie}
\newacronym{uuid}{UUID}{Universally Unique Identifier -- univerzálne jedinečný identifikátor}
\newacronym{ux}{UX}{User Experience -- používateľská skúsenosť}
\newacronym{ynab}{YNAB}{You Need A Budget -- aplikácia pre správu rozpočtu}
